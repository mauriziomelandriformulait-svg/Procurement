function apriGestioneOrdiniDaIframe() { // Accedi alla finestra principale dell'ERP const mainWindow = window.parent; if (mainWindow && mainWindow.angular) { const appElement = mainWindow.document.querySelector('[ng-app]') || mainWindow.document.body; const $injector = mainWindow.angular.element(appElement).injector(); if ($injector) { const workareaService = $injector.get('workareaService'); // Parametri ricavati dal tuo log alla riga: // WA {index: 1, id: 'ORCLI010orcli010r.txt', waId: 'ORCLI010', title: 'Gestione ordini'...} const menuItem = { id: 'ORCLI010orcli010r.txt', waId: 'ORCLI010', title: 'Gestione ordini' }; // Chiamata al servizio del modulo principale mainWindow.angular.element(appElement).scope().$apply(function() { // Sostituisci '.open' con il metodo effettivo di workareaService se fallisce workareaService.open(menuItem); }); } } else { console.error("Impossibile accedere al framework principale."); } }
